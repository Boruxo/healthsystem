public class Jugador {

    String nombre;
    String dañogolpe;
    String ataque;
    String recibirDaño;
    String recibirVida; 
    
    public static void main(String [] args){
        //jugador
        Jugador Jugador1 = new Jugador();
        
        Jugador1.nombre = "Junior Bicicleta Fernandez";
        Jugador1.vida = "100";
        Jugador1.dañogolpe = "12";
        Jugador1.ataque = "12";
        Jugador1.recibirDaño = "13";
        Jugador1.recibirVida = "100";

        System.out.println("El daño del Jugador es: "+Jugador1.dañogolpe);
